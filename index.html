/** * APPS SCRIPT v2 — KHO VẬT TƯ M&R (PIN + XEM KHO) * ------------------------------------------------ * NÂNG CẤP TỪ v1: * - Gửi phiếu phải kèm PIN đúng (kiểm tra phía server). * - doGet trả dữ liệu tồn kho + lịch sử để trang web hiển thị tab "Xem kho". * * CÀI PIN (làm 1 lần): * 1. Trong Apps Script editor: bấm icon bánh răng "Project Settings" (trái). * 2. Kéo xuống "Script Properties" > Add script property: * Property: PIN Value: <6 số Leon tự đặt, vd 250719> * 3. Save. * ĐỔI PIN sau này: sửa Value đó — có hiệu lực NGAY, không cần deploy lại. * * DEPLOY LẠI CODE (bắt buộc vì code đổi): * Deploy > Manage deployments > icon bút ✏️ > Version: New version > Deploy. * URL giữ nguyên — không phải sửa HTML. */ var TAB_NAMES = { "XUAT_DP1": "XUAT_DP1_RESP", "NHAP_DP1": "NHAP_DP1_RESP", "XUAT_DP3": "XUAT_DP3_RESP", "NHAP_DP3": "NHAP_DP3_RESP" }; var HEADERS = ["Timestamp", "Ngày", "Vật tư", "Số lượng", "Ghi chú", "Mã VT", "Đvt"]; var HISTORY_LIMIT = 30; // số dòng lịch sử gần nhất trả về mỗi tab // ============ GHI PHIẾU (có PIN) ============ function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(20000); try { var data = JSON.parse(e.postData.contents); // --- Kiểm tra PIN (server-side) --- var pinStored = PropertiesService.getScriptProperties().getProperty("PIN"); if (!pinStored) { return jsonOut({ status: "error", message: "Chưa cài PIN trong Script Properties. Báo Leon." }); } if (String(data.pin || "") !== String(pinStored)) { return jsonOut({ status: "wrong_pin", message: "Mã PIN không đúng." }); } // --- Kiểm tra dữ liệu phiếu --- if (!data.depot || !data.type || !data.date || !data.items || !data.items.length) { return jsonOut({ status: "error", message: "Thiếu dữ liệu phiếu." }); } var key = String(data.type).toUpperCase() + "_" + String(data.depot).toUpperCase(); var tabName = TAB_NAMES[key]; if (!tabName) { return jsonOut({ status: "error", message: "Tổ/loại phiếu không hợp lệ: " + key }); } var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]).setFontWeight("bold"); sheet.setFrozenRows(1); } var now = new Date(); var rows = []; for (var i = 0; i < data.items.length; i++) { var it = data.items[i]; var qty = Number(it.qty); if (!it.ma || !qty || qty < 1) continue; rows.push([ now, String(data.date), String(it.ten || "") + " (" + String(it.ma) + ")", qty, String(data.note || ""), String(it.ma), String(it.dvt || "") ]); } if (!rows.length) { return jsonOut({ status: "error", message: "Không có dòng vật tư hợp lệ." }); } sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, HEADERS.length).setValues(rows); return jsonOut({ status: "ok", written: rows.length, tab: tabName }); } catch (err) { return jsonOut({ status: "error", message: String(err) }); } finally { lock.releaseLock(); } } // ============ XEM KHO (không cần PIN) ============ function doGet(e) { try { var mode = e && e.parameter ? e.parameter.mode : ""; if (mode !== "view") { return jsonOut({ status: "ok", message: "Apps Script kho vật tư đang hoạt động." }); } var ss = SpreadsheetApp.getActiveSpreadsheet(); // Tồn kho từ 2 tab DANH_MUC: A=Mã, B=Tên, C=Đvt, D=Tồn gốc, E=Nhập, F=Xuất, G=Tồn var stock = {}; ["DP1", "DP3"].forEach(function (dp) { var sh = ss.getSheetByName("DANH_MUC_" + dp); var out = []; if (sh) { var vals = sh.getRange(2, 1, sh.getLastRow() - 1, 7).getValues(); for (var i = 0; i < vals.length; i++) { if (!vals[i][0]) continue; out.push({ ma: String(vals[i][0]), ten: String(vals[i][1]), dvt: String(vals[i][2]), nhap: Number(vals[i][4]) || 0, xuat: Number(vals[i][5]) || 0, ton: Number(vals[i][6]) || 0 }); } } stock[dp] = out; }); // Lịch sử gần nhất từ 4 tab response var history = {}; Object.keys(TAB_NAMES).forEach(function (key) { var sh = ss.getSheetByName(TAB_NAMES[key]); var out = []; if (sh && sh.getLastRow() > 1) { var last = sh.getLastRow(); var start = Math.max(2, last - HISTORY_LIMIT + 1); var vals = sh.getRange(start, 1, last - start + 1, 7).getValues(); for (var i = vals.length - 1; i >= 0; i--) { // mới nhất trước out.push({ ts: vals[i][0] instanceof Date ? Utilities.formatDate(vals[i][0], "GMT+7", "dd/MM HH:mm") : String(vals[i][0]), ngay: String(vals[i][1]), vt: String(vals[i][2]), sl: Number(vals[i][3]) || 0, ghichu: String(vals[i][4] || "") }); } } history[key] = out; }); return jsonOut({ status: "ok", stock: stock, history: history, updated: Utilities.formatDate(new Date(), "GMT+7", "dd/MM/yyyy HH:mm") }); } catch (err) { return jsonOut({ status: "error", message: String(err) }); } } function jsonOut(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
